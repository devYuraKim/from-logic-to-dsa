```text
// dynamic programing (DP)
vector<int> dp(n, INT_MAX) // i th index도달할 수 있는 최소 jump 수
dp[0] = 0;
for (int i = 0; i < n; ++i) {
for (int j = 0; j <= nums[i]; j) {
// [4, 5, 1, 1, 3]
int next = min(i + j, n - 1);  // next [0, 4]
// dp[next] 의 최소 jump, 현재 i에서 1번 뛰어서 갔거나, 아니면 i가 아닌곳에서 뛰어서 도달했을거다
dp[next] = min(dp[next], dp[i] + 1);
}
}

return dp[n - 1];
```
